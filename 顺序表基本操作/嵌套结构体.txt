#include<iostream>
using namespace std;

#define maxsize 10
typedef struct{
	int a;
	char b;
	char c[5];
}elemtype;
typedef struct{
elemtype *elem;
int size;
}sqlist;

void initlist(sqlist &l)
{
	l.elem=new elemtype[maxsize];
	
	if(l.elem) l.size=0;

}

void creatlist(sqlist &l,int m)
{
	
	for(int i=0;i<m;i++)
	{
	cin>>l.elem[i].a;
	cin>>l.elem[i].b;
	cin>>l.elem[i].c;
		l.size++;
	}
}

void show(sqlist &l)
{
	for(int i=0;i<l.size;i++)	
cout<<l.elem[i].a<<l.elem[i].b<<l.elem[i].c<<endl;
}



int main()
{
	sqlist L;
	initlist(L);
	printf("请输入元素（空格分隔）:");
	creatlist(L,5);
    show(L);

	return 0;
}
