#include<stdio.h>
#include<iostream>
using namespace std;
int a[9], n, count;
bool check_2 (int a[ ],int k)
{
    for(int i=1;i<=k-1;i++) 
    {
        if((abs(a[i]-a[k])==k-i)||(a[i]==a[k]))
            return false;
    }      
    return true;
}
void backtrack(int k)
{
    if (k>n)//ур╣╫╫Б
    {
       for(int i=1;i<=8;i++)  
        {
            cout<<a[i];
        }
        cout<<endl;
        count++;
    }
    else
    {
        for (int i = 1;i <=n; i++)
        {
            a[k] = i;
            if (check_2(a,k) ) backtrack(k+1);
        }
    }
    
}

void main()
{
    n=8,count=0;
    backtrack(1);
    cout<<count<<endl;
}
